javascript
Copiar código
// languageLinks.js

document.addEventListener('DOMContentLoaded', () => {
    // Get current page URL
    let currentURL = window.location.href;
    let baseURL = currentURL.replace(/(EN|ES)\.html$/, ''); // Base URL without language suffix
    let languageLinksContainer = document.querySelector('.language-links');

    // Determine the current language
    let isEnglish = currentURL.includes('EN.html');
    let isSpanish = currentURL.includes('ES.html');
    let isCatalan = !isEnglish && !isSpanish;

    // Generate language links
    let linksHTML = '';

    if (isCatalan) {
        // If in Catalan, create links for English and Spanish
        linksHTML += `<a href="${baseURL}EN.html">en</a>`;
        linksHTML += `<a href="${baseURL}ES.html">es</a>`;
    } else if (isEnglish) {
        // If in English, create links for Catalan and Spanish
        linksHTML += `<a href="${baseURL}.html">cat</a>`;
        linksHTML += `<a href="${baseURL}ES.html">es</a>`;
    } else if (isSpanish) {
        // If in Spanish, create links for Catalan and English
        linksHTML += `<a href="${baseURL}.html">cat</a>`;
        linksHTML += `<a href="${baseURL}EN.html">en</a>`;
    }

    // Add links to the language-links container
    if (languageLinksContainer) {
        languageLinksContainer.innerHTML = linksHTML;
    }
});

Generación de enlaces:

Dependiendo del idioma actual, se generan los enlaces para los otros idiomas. Por ejemplo, si estás en la versión en catalán, se crean los enlaces para inglés y español.
Utilizo baseURL.replace(/(EN|ES)\.html$/, '') para obtener la URL base sin sufijos de idioma.
Inserción de los enlaces:

Una vez generados los enlaces, se insertan en el contenedor con la clase .language-links.
Instrucciones de uso:
Asegúrate de que todas las páginas tengan la estructura necesaria, con el contenedor <div class="language-links"></div>.
Incluye el script languageLinks.js en tus páginas HTML antes de cerrar la etiqueta <body>:

2. Enlazar un archivo .css separado
Para usar un archivo CSS externo, utilizas la etiqueta <link> dentro del <head> del archivo HTML. Esto asegura que el CSS se cargue junto con el HTML.

Ejemplo:

html
Copiar código
<link rel="stylesheet" href="styles.css">
rel="stylesheet": Este atributo especifica que el archivo es una hoja de estilo CSS.
href: Este atributo especifica la ruta del archivo CSS, en este caso styles.css. Si el archivo CSS está en el mismo directorio que el HTML, esta ruta es correcta. Si estuviera en otro lugar, deberías ajustar la ruta (e.g., css/styles.css si estuviera dentro de una carpeta llamada css).

document.addEventListener('DOMContentLoaded', () => {
    let currentURL = window.location.href;
    let baseURL = currentURL.replace(/(EN|ES)\.html$/, '');
    let languageLinksContainer = document.querySelector('.language-links');

    let isEnglish = currentURL.includes('EN.html');
    let isSpanish = currentURL.includes('ES.html');
    let isCatalan = !isEnglish && !isSpanish;

    let linksHTML = '';
    if (isCatalan) {
        linksHTML += `<a href="${baseURL}EN.html">en</a>`;
        linksHTML += `<a href="${baseURL}ES.html">es</a>`;
    } else if (isEnglish) {
        linksHTML += `<a href="${baseURL}.html">cat</a>`;
        linksHTML += `<a href="${baseURL}ES.html">es</a>`;
    } else if (isSpanish) {
        linksHTML += `<a href="${baseURL}.html">cat</a>`;
        linksHTML += `<a href="${baseURL}EN.html">en</a>`;
    }

    if (languageLinksContainer) {
        languageLinksContainer.innerHTML = linksHTML;
    }
});
